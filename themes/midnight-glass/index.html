/* Midnight Glass Theme by yx.kxi */

:root {
  --glass-bg: rgba(20, 20, 35, 0.45);
  --glass-border: rgba(255, 255, 255, 0.08);
  --glass-highlight: rgba(255, 255, 255, 0.12);

  --background-primary: #0a0a12;
  --background-secondary: #0f0f18;
  --background-tertiary: #141422;

  --accent-color: #8ab4ff;
  --accent-glow: 0 0 14px rgba(138, 180, 255, 0.45);
}

body {
  background: var(--background-primary) !important;
}

.theme-dark {
  --background-primary: var(--background-primary);
  --background-secondary: var(--background-secondary);
  --background-tertiary: var(--background-tertiary);
  --brand-experiment: var(--accent-color);
}

/* Glass Panels */
.theme-dark .container-1,
.theme-dark .container-2,
.theme-dark .sidebar-1 {
  background: var(--glass-bg);

}

/* Buttons */

  background: var(--accent-color);

}

}

/* Highlights */
